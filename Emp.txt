@override
public int Hash() {
//当两个对象的equals为true时，两个对象的hashcode一定相同

	final int prime = 31;
	int result = 1;
	result = prime * result + id;
	return result;
}

public boolen equals(Object o) {
//自反性：自己和自己比得到结果为true
//永久性：多次比较应该是相同的
//传递性：A和B为true，B和C为true，则A和C为true
	if(this == o) {
		return true;
	}
	if(o == null) {
		return false;
	}
	if(getClass() != o.getClass())
		return false;
	Emp other = (Emp) obj;
	if(id != other.id)
		return false;
	return true;
}
